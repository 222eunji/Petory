// 랜덤 펫 아이콘 배열 const petIcons = ['🐶', '🐱', '🐰', '🐹', '🐦', '🐠', '🐢', '🦔', '🐭', '🐷']; // 사용자 이름을 기반으로 일관된 랜덤 아이콘 생성 function getRandomPetIcon(userName) { // 사용자 이름을 해시코드로 변환하여 일관된 인덱스 생성 let hash = 0; for (let i = 0; i < userName.length; i++) { const char = userName.charCodeAt(i); hash = ((hash << 5) - hash) + char; hash = hash & hash; // 32bit 정수로 변환 } // 항상 양수로 만들고 배열 길이로 나머지 연산 const index = Math.abs(hash) % petIcons.length; return petIcons[index]; }
